####VERCHR

setwd("C:/Users/David Murillo/R/Tesis/Datos/")
getwd()

VERCHR<-read.csv("C:/Users/David Murillo/R/Tesis/Datos/VERCHR.csv")

VERCHR$Sistema = factor(VERCHR$Sistema,
                      levels=unique(VERCHR$Sistema))


library(ggplot2)

qplot(x    = Sistema , 
      y    = Mean,
      data = Sum) +


library(rcompanion)

Sum = groupwiseMean(VERCHR ~ Sistema,
                    data   = VERCHR,
                    conf   = 0.95,
                    digits = 3)

library(FSA)

Sum = Summarize(VERCHR ~ Sistema, 
                data=VERCHR)

Sum$se = Sum$sd / sqrt(Sum$n)

library(ggplot2)

pd = position_dodge(.2)   

ggplot(Sum,                 
       aes(x     = Sistema,
           y     = mean,
           )) +

    geom_point(shape = 15, 
               size  = 4, 
             position = pd) +
    geom_errorbar(aes(ymin  = mean - se,
                      ymax  = mean + se),
                      width = 0.2, 
                      size  = 0.7, 
                      position = pd) +
    theme_bw() +
    theme(axis.title = element_text(face = "bold")) +

    ylab("Abundancia")+
    labs(title = "Abundacia de Vermivora chrysoptera")
